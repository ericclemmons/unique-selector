    lib/user.mjs

    import Fingerprint2 from 'fingerprintjs2';
import Cookies from 'js-cookie';
// import { promisify, } from 'util';
export const cookieName = 'jmlfp';

export class User{
  constructor() {
    const { appCodeName, appName, appVersion, userAgent, cookieEnabled, doNotTrack, language, platform, product, vendor, } = navigator;
    this.user_app_code_name = appCodeName;
    this.user_app_name = appName;
    this.user_app_version = appVersion;
    this.user_cookie_enabled = cookieEnabled;
    this.user_do_not_track = doNotTrack;
    this.user_language = language;
    this.user_platform = platform;
    this.user_product = product;
    this.user_vendor = vendor;
    this.user_agent = userAgent;
  }
  async getFingerprint() {
    if (this.user_fingerprint) return this.user_fingerprint;
    else if (Cookies.get(cookieName)) {
      this.user_fingerprint = Cookies.get(cookieName);
      return this.user_fingerprint;
    } else {
      const components = await Fingerprint2.getPromise();
      const fingerprintString = Fingerprint2.x64hash128(components.map(pair => pair.value).join(), 31);
      Cookies.set(cookieName, fingerprintString);
      this.user_fingerprint = fingerprintString;
      return this.user_fingerprint;
    }
  }
}
